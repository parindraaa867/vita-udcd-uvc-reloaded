# udcd_uvc configuration (optional)
#
# Copy this file to ONE of:
#   ux0:/data/udcd_uvc.txt
#   ur0:/tai/udcd_uvc.txt     (read earliest at boot - most reliable)
#
# Lines are "key = value". '#' or ';' start a comment. If the file is missing,
# the built-in defaults (shown below) are used. The companion app
# (udcd_uvc_config.vpk) writes this file for you.
#
# NOTE: resolution and frame rate are NOT set here - the plugin advertises all
# modes over UVC and you pick them in your capture app (OBS, etc.).

# Blank the Vita screen while a host is capturing. 1 = on, 0 = off.
screen_off = 1

# Allow SELECT + UP to toggle the screen on/off by hand while capturing.
screen_toggle = 1

# Low-latency mode: 1 = grab the freshest frame each cycle (lower latency,
# slightly lower max FPS); 0 = throughput/overlap (default).
low_latency = 0

# Keep the console from auto-suspending while capturing. 1 = on, 0 = off.
prevent_suspend = 1

# Blank the screen after N minutes idle (not capturing). 0 = off.
idle_screen_off = 0

# Seconds to wait at boot before the USB device goes live.
livearea_delay = 15

# All of the above (except livearea_delay) apply live - the config app tells
# the running plugin to re-read this file, no reboot needed.
